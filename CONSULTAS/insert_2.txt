		T013.NHISTOR1,

		T013.APELLID2,
		T013.NOMBRE,
		'|' as I,
		T971.APELLID1 as APELLID1971,
		T971.APELLID2 as APELLID2971,
		T971.NOMBRE as NOMBRE971,

		T971.DOMICILI,
		T013.DOMICILI,
		T971.DNI,

WHERE 
	--T971.SITUACIO='D'
	--AND 
T013.APENOM1 LIKE 'ESPARZA,%';
