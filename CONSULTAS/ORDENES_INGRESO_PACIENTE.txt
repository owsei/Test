
/*ORDENES DE INGRESO Y PACIENTE*/

SELECT T971.APENOM1,T002M.*
FROM SQLEXP.TH002M T002M
INNER JOIN SQLEXP.TH971 T971 ON T002M.NUMEROHISTORIA=T971.NHISTOR1
WHERE 
T002M.TIPOHOSPITALIZACION='D' AND 
T002M.FECHACREACION > TIMESTAMP('2017-01-01') AND
T002M.NUMEROINGRESO =''
ORDER BY FECHACREACION DESC;
